        % Button pushed function: runButton
        function runButtonPushed(app, event)
     
    if isempty(app.UIAxes.Children)
        uialert(app.UIFigure, 'Veuillez d''abord sélectionner une image.', 'Aucune image chargée');
        return;
    end 
    image = app.UIAxes.Children.CData; 
    [h, l, ~] = size(image);
    debutLigne = app.debutdeLigneEditField.Value;  
    finLigne = app.findeLigneEditField.Value;      
    debutColonne = app.debutdeColonneEditField.Value;  
    finColonne = app.findeColonneEditField.Value;

    uialert(app.UIFigure, 'La valeur de ligne est invalide, entrez une autre .', 'Erreur de sélection des lignes');
    return;
    end          
    if debutColonne < 1 || finColonne > l || debutColonne >= finColonne
    uialert(app.UIFigure, 'La valeur de colonne est invalide, entrez une autre.', 'Erreur de sélection des colonnes');
    return;
    end
    bloc = image(debutLigne:finLigne, debutColonne:finColonne, :);
    imshow(bloc, 'Parent', app.UIAxes2);
            
        end

        % Button pushed function: ajouterBruitButton
        function ajouterBruitButtonPushed(app, event)
             if isempty(app.UIAxes.Children)
      uialert(app.UIFigure, 'Veuillez charger une image dans page 1 avant d''ajouter du bruit.', 'Aucune image chargée');
      return;
             end
             image = app.UIAxes.Children.CData;
              bruit = app.typedebruitDropDown.Value; 
              co = app.coefficientdebruitageEditField.Value; 
             if co < 0 || co > 1
     uialert(app.UIFigure, 'Le coefficient de bruitage doit être entre 0 et 1.', 'Valeur de bruit invalide');
     return;
             end
              switch bruit
        case 'Salt&pepper'
            i = imnoise(image, 'salt & pepper', co);
        case 'Gaussian'
            i = imnoise(image, 'gaussian', 0, co);
        case 'Speckle'
           i = imnoise(image, 'speckle', co);
        otherwise
            uialert(app.UIFigure, 'Type de bruit non reconnu.', 'Erreur');
            return;
              end
    imshow(i, 'Parent', app.UIAxes6);
        end
        % Button pushed function: ajouterfiltreButton
        function ajouterfiltreButtonPushed(app, event)
            
             tailleFiltre = app.saisirlatailledufiltreEditField.Value;
              hauteurImage = app.hauteurdelimageEditField.Value;
              largeurImage = app.largeurdelimageEditField.Value;
               if isempty(app.UIAxes6.Children)
       uialert(app.UIFigure, 'Aucune image n''a été trouvée.', 'Erreur');
       return;
               end
              ImageSelectionnee = app.UIAxes6.Children.CData;
             if isempty(ImageSelectionnee)
    uialert(app.UIFigure, 'Veuillez sélectionner une image dans axe 6.', 'Erreur');
    return;
             end
            if tailleFiltre > min(hauteurImage, largeurImage)
    uialert(app.UIFigure, 'La taille du filtre est trop grande pour l''image.', 'Erreur');
    return;
            elseif mod(tailleFiltre, 2) == 0
    uialert(app.UIFigure, 'La taille du filtre doit être impaire (ex : 3, 5, 7, etc.)', 'Erreur');
    return;
            end
               switch app.typedefiltreDropDown.Value
     case 'Medium'
         if ndims(ImageSelectionnee) == 3
             imageFiltree(:,:,1) = medfilt2(ImageSelectionnee(:,:,1), [tailleFiltre tailleFiltre]);
             imageFiltree(:,:,2) = medfilt2(ImageSelectionnee(:,:,2), [tailleFiltre tailleFiltre]);
             imageFiltree(:,:,3) = medfilt2(ImageSelectionnee(:,:,3), [tailleFiltre tailleFiltre]);
         else
            imageFiltree = medfilt2(ImageSelectionnee, [tailleFiltre tailleFiltre]);
        end
    case 'Moyen'
        h = fspecial('average', [tailleFiltre tailleFiltre]);
        imageFiltree = imfilter(ImageSelectionnee, h);
    case 'Laplacien'
        h = fspecial('laplacian');
        imageFiltree = imfilter(ImageSelectionnee, h);
    otherwise
        uialert(app.UIFigure, 'Type de filtre inconnu.', 'Erreur');
        return;
               end
           
            imshow(imageFiltree, 'Parent', app.UIAxes8);
            
        end
        % Button pushed function: calculerPSNRButton
        function calculerPSNRButtonPushed(app, event)
            
    imgOriginale = getimage(app.UIAxes);  
       
    imgFiltree = getimage(app.UIAxes8);  
    
    
    mse = immse(imgFiltree, imgOriginale);
    
    psnrValue = 10 * log10(255^2 / mse);
    
    app.PSNREditField.Value = psnrValue;
        end
        % Callback function
        function calculerPSNRButton_2Pushed(app, event)
             
            imgOriginale = getimage(app.UIAxes);  
            
            imgBruitee = getimage(app.UIAxes6);  
            
            mseValue = immse(imgBruitee, imgOriginale);
            
            psnrValue = 10 * log10(255^2 / mseValue);
            
           
            app.PSNREditField_2.Value = num2str(psnrValue);  
            
        end
    end

        % Button pushed function: selectionneruneImageButton
        function selectionneruneImageButtonPushed(app, event)
            
            [file, path] = uigetfile({'*.jpg;*.png;*.bmp;*.tif', 'Image Files (*.jpg, *.png, *.bmp, *.tif)'});
    if isequal(file, 0)
        disp('Aucune image sélectionnée.');
    else
        pathedelimage = fullfile(path, file);
        image = imread(pathedelimage); 
        imshow(image, 'Parent', app.UIAxes); 
        app.hauteurdelimageEditField.Value = size(image, 1); 
        app.largeurdelimageEditField.Value = size(image, 2); 
    end
    
        end
        % Button pushed function: runButton
        function runButtonPushed(app, event)
     
    if isempty(app.UIAxes.Children)
        uialert(app.UIFigure, 'Veuillez d''abord sélectionner une image.', 'Aucune image chargée');
        return;
    end 
    image = app.UIAxes.Children.CData; 
    [h, l, ~] = size(image);
    debutLigne = app.debutdeLigneEditField.Value;  
    finLigne = app.findeLigneEditField.Value;      
    debutColonne = app.debutdeColonneEditField.Value;  
    finColonne = app.findeColonneEditField.Value;      
    if debutLigne < 1 || finLigne > h || debutLigne >= finLigne
    uialert(app.UIFigure, 'La valeur de ligne est invalide, entrez une autre .', 'Erreur de sélection des lignes');
    return;
    end          
    if debutColonne < 1 || finColonne > l || debutColonne >= finColonne
    uialert(app.UIFigure, 'La valeur de colonne est invalide, entrez une autre.', 'Erreur de sélection des colonnes');
    return;
    end
    bloc = image(debutLigne:finLigne, debutColonne:finColonne, :);
    imshow(bloc, 'Parent', app.UIAxes2);
            
        end
        % Button pushed function: runButton_2
        function chercherniveaudegrisButtonPushed(app, event)
        
    if isempty(app.UIAxes.Children)
        uialert(app.UIFigure, 'Veuillez d''abord sélectionner une image dans tab1.', 'Aucune image chargée');
        return;
    end
     image = app.UIAxes.Children.CData;
    if size(image, 3) == 3
        g = rgb2gray(image);

        g = image;

     ng = app.niveaudegrisEditField.Value;  
         if ng < 0 || ng > 255

        m = sum(g(:) == ng);%m freqence de nv de gris dans imgg
        app.nombredapparitionEditField.Value = m;
        imgscannee = 255 * ones(size(g), 'uint8');
        imgscannee(g == ng) = ng;
        imshow(imgscannee, 'Parent', app.UIAxes3);  
        end

        function ajouterBruitButtonPushed(app, event)
             if isempty(app.UIAxes.Children)
      uialert(app.UIFigure, 'Veuillez charger une image dans page 1 avant d''ajouter du bruit.', 'Aucune image chargée');
      return;
             end
             image = app.UIAxes.Children.CData;
              bruit = app.typedebruitDropDown.Value; 
              co = app.coefficientdebruitageEditField.Value; 
             if co < 0 || co > 1
     uialert(app.UIFigure, 'Le coefficient de bruitage doit être entre 0 et 1.', 'Valeur de bruit invalide');
     return;
             end
              switch bruit
        case 'Salt&pepper'
            i = imnoise(image, 'salt & pepper', co);
        case 'Gaussian'
            i = imnoise(image, 'gaussian', 0, co);

           i = imnoise(image, 'speckle', co);
        otherwise
            uialert(app.UIFigure, 'Type de bruit non reconnu.', 'Erreur');
            return;
              end
    imshow(i, 'Parent', app.UIAxes6);

        % Button pushed function: ajouterfiltreButton
        function ajouterfiltreButtonPushed(app, event)
            
             tailleFiltre = app.saisirlatailledufiltreEditField.Value;
              hauteurImage = app.hauteurdelimageEditField.Value;
              largeurImage = app.largeurdelimageEditField.Value;
               if isempty(app.UIAxes6.Children)
       uialert(app.UIFigure, 'Aucune image n''a été trouvée.', 'Erreur');
       return;
               end
              ImageSelectionnee = app.UIAxes6.Children.CData;
             if isempty(ImageSelectionnee)
    uialert(app.UIFigure, 'Veuillez sélectionner une image dans axe 6.', 'Erreur');
    return;
             end
            if tailleFiltre > min(hauteurImage, largeurImage)
    uialert(app.UIFigure, 'La taille du filtre est trop grande pour l''image.', 'Erreur');
    return;
            elseif mod(tailleFiltre, 2) == 0
    uialert(app.UIFigure, 'La taille du filtre doit être impaire (ex : 3, 5, 7, etc.)', 'Erreur');
    return;
            end
               switch app.typedefiltreDropDown.Value
     case 'Medium'
         if ndims(ImageSelectionnee) == 3
             imageFiltree(:,:,1) = medfilt2(ImageSelectionnee(:,:,1), [tailleFiltre tailleFiltre]);
             imageFiltree(:,:,2) = medfilt2(ImageSelectionnee(:,:,2), [tailleFiltre tailleFiltre]);
             imageFiltree(:,:,3) = medfilt2(ImageSelectionnee(:,:,3), [tailleFiltre tailleFiltre]);
         else
            imageFiltree = medfilt2(ImageSelectionnee, [tailleFiltre tailleFiltre]);
        end
    case 'Moyen'
        h = fspecial('average', [tailleFiltre tailleFiltre]);
        imageFiltree = imfilter(ImageSelectionnee, h);
    case 'Laplacien'
        h = fspecial('laplacian');
        imageFiltree = imfilter(ImageSelectionnee, h);
    otherwise
        uialert(app.UIFigure, 'Type de filtre inconnu.', 'Erreur');
        return;
               end
           
            imshow(imageFiltree, 'Parent', app.UIAxes8);
            
        end

        end
